#!/bin/sh
### BEGIN INIT INFO
# Provides: geoserver
# Required-Start: $local_fs $remote_fs $network $syslog
# Required-Stop: $local_fs $remote_fs $network $syslog
# Should-Start: postgresql
# Default-Start: 2 3 4 5
# Default-Stop: 0 1 6
# Short-Description: Management of JBoss AS
### END INIT INFO

#Defining directories
JAVA_HOME=/opt/jdk1.7.0_71
export GEOSERVER_HOME=/opt/geoserver-2.6.0
export JAVA_HOME=/opt/jdk1.7.0_71
export GEOSERVER_DATA_DIR=/home/geoserver/data_dir
export GEOSERVER_GEOJSON_LEGACY_CRS=true

case "$1" in
start)
echo "Starting Geoserver"
start-stop-daemon --start --quiet --background --chuid geoserver --exec ${GEOSERVER_HOME}/bin/startup.sh
;;
stop)
echo "Stopping Geoserver"
start-stop-daemon --start --quiet --background --chuid geoserver --exec ${GEOSERVER_HOME}/bin/shutdown.sh -- --connect command=:shutdown
;;
*)
echo "Usage: /etc/init.d/geoserver {start|stop}"
exit 1
;; esac
exit 0
